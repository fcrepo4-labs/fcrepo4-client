017package org.fcrepo.client;
018
019import java.io.InputStream;
020import java.util.Collection;
021import java.util.Date;
022import java.util.Iterator;
023
024import com.hp.hpl.jena.graph.Triple;
025
026/**
027 * Superclass of Fedora Objects and Datastreams containing common functionality.
028 * @author escowles
029 * @since 2014-08-01
030**/
031public interface FedoraResource {
032
033    /**
034     * Copy this Resource to a new path.
035     * @param destination The path of the new copy.
036    **/
037    public void copy( String destination ) throws FedoraException;
038
039    /**
040     * Remove this Resource.
041    **/
042    public void delete() throws FedoraException;
043
044    /**
045     * Get the creation date of this Resource.
046    **/
047    public Date getCreatedDate() throws FedoraException;
048
049    /**
050     * Get the ETag of this Resource.
051    **/
052    public String getEtagValue() throws FedoraException;
053
054    /**
055     * Get the modification date of this Resource.
056    **/
057    public Date getLastModifiedDate() throws FedoraException;
058
059    /**
060     * Get the mixins assigned to this Resource.
061    **/
062    public Collection<String> getMixins() throws FedoraException;
063
064    /**
065     * Get the name of this Resource.
066    **/
067    public String getName() throws FedoraException;
068
069    /**
070     * Get the full path of the Resource, relative to the repository root.
071    **/
072    public String getPath() throws FedoraException;
073
074    /**
075     * Get the RDF properties of this Resource.
076    **/
077    public Iterator<Triple> getProperties() throws FedoraException;
078
079    /**
080     * Get the size of this Resource in bytes.
081    **/
082    public Long getSize() throws FedoraException;
083
084    /**
085     * Move this Resource to a new path.
086     * @param destination The path of the new copy.
087    **/
088    public void move( String destination ) throws FedoraException;
089
090    /**
091     * Update the properties of this Resource using SPARQL Update.
092     * @param sparqlUpdate SPARQL Update command.
093    **/
094    public void updateProperties( String sparqlUpdate ) throws FedoraException;
095
096    /**
097     * Update the properties of this Resource with the provided RDF.
098     * @param updatedProperties RDF properties as an InputStream.
099     * @param contentType Content type of the RDF in updatedProperties (e.g.,
100     *         "text/rdf+n3" or "application/rdf+xml").
101    **/
102    public void updateProperties( InputStream updatedProperties,
103                                  String contentType ) throws FedoraException;
104
105    /**
106     * Check whether this Resource is writable.
107    **/
108    public boolean isWritable();
109}
